%========================
% formatting.tex
%========================
% ONLY formatting
% No package loading here

%----------------------------------------
% Compact line spacing
%----------------------------------------
\setstretch{1.05}

%----------------------------------------
% Paragraph formatting
%----------------------------------------
\setlength{\parindent}{0pt}
\setlength{\parskip}{0.4em}

%----------------------------------------
% Compact equations
%----------------------------------------
\setlength{\abovedisplayskip}{8pt}
\setlength{\belowdisplayskip}{8pt}

%----------------------------------------
% Compact figures/tables spacing
%----------------------------------------
\setlength{\textfloatsep}{10pt}
\setlength{\floatsep}{8pt}
\setlength{\intextsep}{8pt}

%----------------------------------------
% Compact lists
%----------------------------------------
\setlist{
    topsep=3pt,
    itemsep=2pt,
    parsep=0pt
}

%----------------------------------------
% Centered chapter titles
%----------------------------------------
\titleformat{\chapter}[display]
{\centering\bfseries\Huge}
{\chaptername\ \thechapter}
{0.5em}
{}
[
\rule{\textwidth}{1pt}
]

%----------------------------------------
% Compact chapter spacing
%----------------------------------------
\titlespacing*{\chapter}
{0pt}
{-20pt}
{20pt}

%----------------------------------------
% Section formatting
%----------------------------------------
\titleformat{\section}
{\bfseries\Large}
{\thesection}
{0.5em}
{}

\titleformat{\subsection}
{\bfseries\large}
{\thesubsection}
{0.5em}
{}

%----------------------------------------
% Header/footer style
%----------------------------------------

% Slightly taller header
\setlength{\headheight}{14pt}

\pagestyle{fancy}

\fancyhf{}

%----------------------------------------
% Page number
%----------------------------------------
\fancyfoot[C]{\small\thepage}

%----------------------------------------
% Headers
%----------------------------------------
\fancyhead[L]{\small\itshape\nouppercase{\leftmark}}
\fancyhead[R]{\small\nouppercase{\rightmark}}

%----------------------------------------
% Lines
%----------------------------------------
\renewcommand{\headrulewidth}{0.3pt}
\renewcommand{\footrulewidth}{0.3pt}

%----------------------------------------
% Caption formatting
%----------------------------------------
\DeclareCaptionFormat{hanging}{
    #1#2\hspace{0.5em}#3
}

\captionsetup{
    format=hanging,
    labelfont=bf
}

%----------------------------------------
% Equation numbering by chapter
%----------------------------------------
\numberwithin{equation}{chapter}

%----------------------------------------
% Listings style
%----------------------------------------
\lstset{
    basicstyle=\ttfamily\small,
    breaklines=true,
    frame=single
}

%----------------------------------------
% Theorem environments
%----------------------------------------
\theoremstyle{plain}
\newtheorem{theorem}{Theorem}[chapter]
\newtheorem{lemma}[theorem]{Lemma}

\theoremstyle{definition}
\newtheorem{definition}[theorem]{Definition}

%----------------------------------------
% Useful commands
%----------------------------------------
\newcommand{\R}{\mathbb{R}}
\newcommand{\C}{\mathbb{C}}
\newcommand{\Z}{\mathbb{Z}}

%----------------------------------------
% Prevent ugly line breaks
%----------------------------------------
\clubpenalty=10000
\widowpenalty=10000
\displaywidowpenalty=10000

%----------------------------------------
% Chapter quotes
%----------------------------------------

% Usage:
%
% \chapterquote{Quote text}{Author}
%
% Place it immediately after \chapter{...}

\newcommand{\chapterquote}[2]{
    \begin{flushright}
        \begin{minipage}{0.7\textwidth}
            \itshape
            ``#1''
            
            \vspace{0.5em}
            
            \raggedleft
            --- #2
        \end{minipage}
    \end{flushright}
    
    \vspace{1em}
    
    \rule{\textwidth}{1pt}
    
    \vspace{2em}
}
